import os\n    
        
        
        "import shutil\n    
        
        
        "import zipfile\n    
        
        
        "import tarfile\n    
        
        
        "import gzip\n    
        
        
        "import bz2\n    
        
        
        "import sys\n    
        
        
        "import logging\n    
        
        
        "from pathlib import Path\n    
        
        
        "# Set up logging\n    
        
        
        "log_file_path = '/private/var/mobile/Containers/Data/Application/5A3A90C0-A077-4A2C-80A6-0DF006769AC6/Documents/script.log'\n    
        
        
        "logging.basicConfig(filename=log_file_path, level=logging.DEBUG, \n    
        
        
        "                    format='%(asctime)s - %(levelname)s - %(message)s')\n    
        
        
        "def extract_archive(archive_path, extract_dir):\n    
        
        
        "    """Extracts files from various archive formats."""\n    
        
        
        "    logging.info(f'Starting extraction for: {archive_path}')\n    
        
        
        "    try:\n    
        
        
        "        if zipfile.is_zipfile(archive_path):\n    
        
        
        "            with zipfile.ZipFile(archive_path, 'r') as zip_ref:\n    
        
        
        "                zip_ref.extractall(extract_dir)\n    
        
        
        "        elif tarfile.is_tarfile(archive_path):\n    
        
        
        "            with tarfile.open(archive_path, 'r') as tar_ref:\n    
        
        
        "                tar_ref.extractall(extract_dir)\n    
        
        
        "        elif archive_path.endswith('.gz') and not tarfile.is_tarfile(archive_path):\n    
        
        
        "            with gzip.open(archive_path, 'rb') as gz_file:\n    
        
        
        "                with open(archive_path[:-3], 'wb') as output_file:  # Remove .gz extension\n    
        
        
        "                    shutil.copyfileobj(gz_file, output_file)\n    
        
        
        "        elif archive_path.endswith('.bz2') and not tarfile.is_tarfile(archive_path):\n    
        
        
        "            with bz2.open(archive_path, 'rb') as bz2_file:\n    
        
        
        "                with open(archive_path[:-4], 'wb') as output_file:  # Remove .bz2 extension\n    
        
        
        "                    shutil.copyfileobj(bz2_file, output_file)\n    
        
        
        "        else:\n    
        
        
        "            logging.warning(f'Unsupported archive format: {archive_path}')\n    
        
        
        "            print(f"Unsupported archive format: {archive_path}")\n    
        
        
        "    except Exception as e:\n    
        
        
        "        logging.error(f'Failed to extract {archive_path}: {e}', exc_info=True)\n    
        
        
        "        print(f"Failed to extract {archive_path}: {e}")\n    
        
        
        "def process_files(root_path, rar_folder):\n    
        
        
        "    """Recursively processes files and folders, extracting archives and moving RAR files."""\n    
        
        
        "    logging.info(f'Starting to process files in: {root_path}')\n    
        
        
        "    for dirpath, _, filenames in os.walk(root_path):\n    
        
        
        "        for filename in filenames:\n    
        
        
        "            file_path = os.path.join(dirpath, filename)\n    
        
        
        "            if os.path.isfile(file_path):\n    
        
        
        "                if file_path.endswith('.rar'):\n    
        
        
        "                    destination_path = os.path.join(rar_folder, filename)\n    
        
        
        "                    try:\n    
        
        
        "                        shutil.move(file_path, destination_path)\n    
        
        
        "                        logging.info(f'Moved RAR file {file_path} to {destination_path}')\n    
        
        
        "                    except Exception as e:\n    
        
        
        "                        logging.error(f'Failed to move {file_path} to {destination_path}: {e}', exc_info=True)\n    
        
        
        "                        print(f"Failed to move {file_path}: {e}")\n    
        
        
        "                else:\n    
        
        
        "                    extract_archive(file_path, dirpath)\n    
        
        
        "def move_to_cat_folder(root_path):\n    
        
        
        "    """Moves all files and folders to a new 'CAT' folder at the same level as the root path."""\n    
        
        
        "    logging.info(f'Starting to move files to CAT folder from: {root_path}')\n    
        
        
        "    root_path = Path(root_path).resolve()\n    
        
        
        "    cat_folder_path = root_path / 'CAT'  # Create 'CAT' folder at the same level\n    
        
        
        "    cat_folder_path.mkdir(exist_ok=True)\n    
        
        
        "    \n    
        
        
        "    for item in root_path.iterdir():\n    
        
        
        "        if item.name in ['CAT', 'CAT-RAR']:\n    
        
        
        "            continue\n    
        
        
        "        destination_path = cat_folder_path / item.name\n    
        
        
        "        try:\n    
        
        
        "            shutil.move(str(item), destination_path)\n    
        
        
        "            logging.info(f'Moved {item} to {destination_path}')\n    
        
        
        "        except Exception as e:\n    
        
        
        "            logging.error(f'Failed to move {item} to {destination_path}: {e}', exc_info=True)\n    
        
        
        "            print(f"Failed to move {item}: {e}")\n    
        
        
        "if __name__ == "__main__":\n    
        
        
        "    if len(sys.argv) != 2:\n    
        
        
        "        print("Usage: python script.py <path>")\n    
        
        
        "        logging.error('Incorrect usage. Path not provided.')\n    
        
        
        "        sys.exit(1)\n    
        
        
        "    \n    
        
        
        "    target_path = sys.argv[1]\n    
        
        
        "    \n    
        
        
        "    if os.path.exists(target_path):\n    
        
        
        "        logging.info(f'Processing started for path: {target_path}')\n    
        
        
        "        rar_folder_path = os.path.join(target_path, 'CAT-RAR')\n    
        
        
        "        os.makedirs(rar_folder_path, exist_ok=True)\n    
        
        
        "        \n    
        
        
        "        process_files(target_path, rar_folder_path)\n    
        
        
        "        move_to_cat_folder(target_path)\n    
        
        
        "        print("Processing and moving completed!")\n    
        
        
        "        logging.info('Processing completed successfully.')\n    
        
        
        "    else:\n

        "import click\n    
        
        
        "@click.command()\n    
        
        
        "@click.argument('file_path')\n    
        
        
        "def main(file_path):\n    
        
        
        "    # Use subprocess to call the files-to-prompt command\n
