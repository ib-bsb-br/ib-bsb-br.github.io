"
#!/bin/sh\n    
        
        
        "#\n    
        
        
        "# /etc/X11/Xsession\n    
        
        
        "#\n    
        
        
        "# global Xsession file -- used by display managers and xinit (startx)\n    
        
        
        "\n    
        
        
        "# $Id: Xsession 967 2005-12-27 07:20:55Z dnusinow $\n    
        
        
        "\n    
        
        
        "set -e\n    
        
        
        "\n    
        
        
        "PROGNAME=Xsession\n    
        
        
        "\n    
        
        
        "message () {\n    
        
        
        "  # pretty-print messages of arbitrary length; use xmessage if it\n    
        
        
        "  # is available and $DISPLAY is set\n    
        
        
        "  MESSAGE="$PROGNAME: $*"\n    
        
        
        "  echo "$MESSAGE" | fold -s -w ${COLUMNS:-80} >&2\n    
        
        
        "  if [ -n "$DISPLAY" ] && which xmessage > /dev/null 2>&1; then\n    
        
        
        "    echo "$MESSAGE" | fold -s -w ${COLUMNS:-80} | xmessage -center -file -\n    
        
        
        "  fi\n    
        
        
        "}\n    
        
        
        "\n    
        
        
        "message_nonl () {\n    
        
        
        "  # pretty-print messages of arbitrary length (no trailing newline); use\n    
        
        
        "  # xmessage if it is available and $DISPLAY is set\n    
        
        
        "  MESSAGE="$PROGNAME: $*"\n    
        
        
        "  echo -n "$MESSAGE" | fold -s -w ${COLUMNS:-80} >&2;\n    
        
        
        "  if [ -n "$DISPLAY" ] && which xmessage > /dev/null 2>&1; then\n    
        
        
        "    echo -n "$MESSAGE" | fold -s -w ${COLUMNS:-80} | xmessage -center -file -\n    
        
        
        "  fi\n    
        
        
        "}\n    
        
        
        "\n    
        
        
        "errormsg () {\n    
        
        
        "  # exit script with error\n    
        
        
        "  message "$*"\n    
        
        
        "  exit 1\n    
        
        
        "}\n    
        
        
        "\n    
        
        
        "internal_errormsg () {\n    
        
        
        "  # exit script with error; essentially a "THIS SHOULD NEVER HAPPEN" message\n    
        
        
        "  # One big call to message() for the sake of xmessage; if we had two then\n    
        
        
        "  # the user would have dismissed the error we want reported before seeing the\n    
        
        
        "  # request to report it.\n    
        
        
        "  errormsg "$*" \\n    
        
        
        "           "Please report the installed version of the \"x11-common\"" \\n    
        
        
        "           "package and the complete text of this error message to" \\n    
        
        
        "           "<debian-x@lists.debian.org>."\n    
        
        
        "}\n    
        
        
        "\n    
        
        
        "# initialize variables for use by all session scripts\n    
        
        
        "\n    
        
        
        "OPTIONFILE=/etc/X11/Xsession.options\n    
        
        
        "\n    
        
        
        "SYSRESOURCES=/etc/X11/Xresources\n    
        
        
        "USRRESOURCES=$HOME/.Xresources\n    
        
        
        "\n    
        
        
        "SYSSESSIONDIR=/etc/X11/Xsession.d\n    
        
        
        "USERXSESSION=$HOME/.xsession\n    
        
        
        "USERXSESSIONRC=$HOME/.xsessionrc\n    
        
        
        "ALTUSERXSESSION=$HOME/.Xsession\n    
        
        
        "ERRFILE=$HOME/.xsession-errors\n    
        
        
        "\n    
        
        
        "# attempt to create an error file; abort if we cannot\n    
        
        
        "if (umask 077 && touch "$ERRFILE") 2> /dev/null && [ -w "$ERRFILE" ] &&\n    
        
        
        "  [ ! -L "$ERRFILE" ]; then\n    
        
        
        "  chmod 600 "$ERRFILE"\n    
        
        
        "elif ERRFILE=$(tempfile 2> /dev/null); then\n    
        
        
        "  if ! ln -sf "$ERRFILE" "${TMPDIR:=/tmp}/xsession-$USER"; then\n    
        
        
        "    message "warning: unable to symlink \"$TMPDIR/xsession-$USER\" to" \\n    
        
        
        "             "\"$ERRFILE\"; look for session log/errors in" \\n    
        
        
        "             "\"$TMPDIR/xsession-$USER\"."\n    
        
        
        "  fi\n    
        
        
        "else\n    
        
        
        "  errormsg "unable to create X session log/error file; aborting."\n    
        
        
        "fi\n    
        
        
        "\n    
        
        
        "exec >>"$ERRFILE" 2>&1\n    
        
        
        "\n    
        
        
        "echo "$PROGNAME: X session started for $LOGNAME at $(date)"\n    
        
        
        "\n    
        
        
        "# sanity check; is our session script directory present?\n    
        
        
        "if [ ! -d "$SYSSESSIONDIR" ]; then\n    
        
        
        "  errormsg "no \"$SYSSESSIONDIR\" directory found; aborting."\n    
        
        
        "fi\n    
        
        
        "\n    
        
        
        "# Attempt to create a file of non-zero length in /tmp; a full filesystem can\n    
        
        
        "# cause mysterious X session failures.  We do not use touch, :, or test -w\n    
        
        
        "# because they won't actually create a file with contents.  We also let standard\n    
        
        
        "# error from tempfile and echo go to the error file to aid the user in\n    
        
        
        "# determining what went wrong.\n    
        
        
        "WRITE_TEST=$(tempfile)\n    
        
        
        "if ! echo "*" >>"$WRITE_TEST"; then\n    
        
        
        "  message "warning: unable to write to ${WRITE_TEST%/*}; X session may exit" \\n    
        
        
        "          "with an error"\n    
        
        
        "fi\n    
        
        
        "rm -f "$WRITE_TEST"\n    
        
        
        "\n    
        
        
        "# use run-parts to source every file in the session directory; we source\n    
        
        
        "# instead of executing so that the variables and functions defined above\n    
        
        
        "# are available to the scripts, and so that they can pass variables to each\n    
        
        
        "# other\n    
        
        
        "SESSIONFILES=$(run-parts --list $SYSSESSIONDIR)\n    
        
        
        "if [ -n "$SESSIONFILES" ]; then\n    
        
        
        "  set +e\n    
        
        
        "  for SESSIONFILE in $SESSIONFILES; do\n    
        
        
        "    . $SESSIONFILE\n    
        
        
        "  done\n    
        
        
        "  set -e\n    
        
        
        "fi\n    
        
        
        "\n    
        
        
        "exec ratpoison\n    
        
        
        "\n    
        
        
        "exit 0\n    
        
        
        "\n    
        
        
        "# vim:set ai et sts=2 sw=2 tw=80:\n    
        
        
        "
\n "
