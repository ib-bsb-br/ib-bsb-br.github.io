        "def getTxt(filepath, index=0) -> str:  # get .txt content\n    
        
        
        "    coding = ("utf-8", "gbk", "utf-16")  # codings\n    
        
        
        "    if 0 <= index < len(coding):  # in the range\n    
        
        
        "        try:\n    
        
        
        "            with open(filepath, "r", encoding=coding[index]) as f:\n    
        
        
        "                content = f.read()\n    
        
        
        "            return content[1:] if content.startswith("\ufeff") else content  # if utf-8 with BOM, remove BOM\n    
        
        
        "        except (UnicodeError, UnicodeDecodeError):\n    
        
        
        "            return getTxt(filepath, index + 1)  # recursion\n    
        
        
        "        except:\n    
        
        
        "            return None\n    
        
        
        "    else:\n    
        
        
        "        return None  # out of range\n    
        
        
        "\n    
        
        
        "def removeCommentLine(text) -> str:  # remove comment lines\n    
        
        
        "    lines = text.split("\n")\n    
        
        
        "    for i, line in enumerate(lines):\n    
        
        
        "        for j in range(len(line)):\n    
        
        
        "            if line[j] == "%" and (0 == j or line[j - 1] != "\\"):\n    
        
        
        "                lines[i] = lines[i][:j]\n    
        
        
        "    return "\n".join(lines)\n

        "def citationSurvey(texFilepaths, isDebug=False) -> dict:\n    
        
        
        "    clearScreen()\n    
        
        
        "    if type(texFilepaths) not in (tuple, list) or not texFilepaths:\n    
        
        
        "        print("As no tex files are found, the checking cannot work. ")\n    
        
        
        "        print("Please press any key to go back. ")\n    
        
        
        "        press_any_key_to_continue()\n    
        
        
        "        return None\n    
        
        
        "\n    
        
        
        "    dicts = {}\n    
        
        
        "    for texFilepath in texFilepaths:\n    
        
        
        "        text = getTxt(texFilepath)\n    
        
        
        "        if text is None:\n    
        
        
        "            print("Read tex file \"{0}\" failed. ".format(texFilepath))\n    
        
        
        "        else:\n    
        
        
        "            lines = removeCommentLine(text).split("\n")\n    
        
        
        "            section = ""\n    
        
        
        "            for line in lines:\n    
        
        
        "                if len(findall("\\\\section\\{.+?\\}", line)):\n    
        
        
        "                    section = findall("\\\\section\\{.+?\\}", line)[0][9:-1]\n    
        
        
        "                elif len(findall("\\\\section\\*\\{.+?\\}", line)):\n    
        
        
        "                    section = findall("\\\\section\\*\\{.+?\\}", line)[0][10:-1]\n    
        
        
        "                elif len(findall("\\\\chapter\\{.+?\\}", line)):\n    
        
        
        "                    section = findall("\\\\chapter\\{.+?\\}", line)[0][9:-1]\n    
        
        
        "                elif len(findall("\\\\chapter\\*\\{.+?\\}", line)):\n    
        
        
        "                    section = findall("\\\\chapter\\*\\{.+?\\}", line)[0][10:-1]\n    
        
        
        "                if not section:\n    
        
        
        "                    continue\n    
        
        
        "                for cite in findall("\\\\cite\\{.+?\\}", line):\n    
        
        
        "                    cite = cite[cite.index("{") + 1:-1]\n    
        
        
        "                    for subCite in cite.split(","):\n    
        
        
        "                        dicts.setdefault(section, {})\n    
        
        
        "                        dicts[section].setdefault(subCite.strip(), 0)\n    
        
        
        "                        dicts[section][subCite.strip()] += 1\n    
        
        
        "\n    
        
        
        "    for section in list(dicts.keys()):\n    
        
        
        "        totalCount = 0\n    
        
        
        "        for subCite in list(dicts[section].keys()):\n    
        
        
        "            totalCount += dicts[section][subCite]\n    
        
        
        "        dicts[section] = {\n    
        
        
        "            "citation_count": len(dicts[section]),\n    
        
        
        "            "total_count": totalCount,\n    
        
        
        "            "details": dicts[section],\n    
        
        
        "        }\n    
        
        
        "\n    
        
        
        "    if isDebug:\n    
        
        
        "        print("dicts =", dicts)\n    
        
        
        "\n    
        
        
        "    print("This is the citation survey. The result is as follows. ")\n    
        
        
        "    print()\n    
        
        
        "    if len(dicts):\n    
        
        
        "        sections = sorted(list(dicts.keys()))\n    
        
        
        "        maxSectionLen = max([len(section) for section in sections])\n    
        
        
        "        for section in sections:\n    
        
        
        "            print(section.ljust(maxSectionLen + 2) + "contains {0} citations with {1} citation count(s). ".format(dicts[section]["citation_count"], dicts[section]["total_count"]))\n    
        
        
        "    else:\n    
        
        
        "        print("No citation is found. ")\n    
        
        
        "\n
