#!/bin/bash
set -euo pipefail

# install-procfd-rk3588-debian11.sh
# Purpose: Build and install procfd (Rust lsof-like tool) on Debian 11 (Bullseye) ARM64 (RK3588)
# Run context: Execute as a normal user from within ~/ (or a subdir under it). Requires sudo.
#
# Safety & integrity:
#   - No destructive operations without explicit confirmation.
#   - Explicit confirmation is required before executing any downloaded installer binary.
#   - If /usr/local/bin/procfd exists, confirmation is required before overwriting it.
#
# Connectivity:
#   - If the source directory does not exist, the script can clone from GitHub (default).
#   - Disable cloning with --no-clone for offline environments.

# ----------------------------
# Configuration (edit if needed)
# ----------------------------
SRC_DIR_DEFAULT="$HOME/deshaw-procfd"
SRC_REPO_URL="https://github.com/deshaw/procfd.git"
INSTALL_PATH="/usr/local/bin/procfd"

RUSTUP_INIT_URL_AARCH64="https://static.rust-lang.org/rustup/dist/aarch64-unknown-linux-gnu/rustup-init"
RUST_2021_MIN_MAJOR=1
RUST_2021_MIN_MINOR=56

# Script options (overridable by flags)
SRC_DIR="$SRC_DIR_DEFAULT"
DO_CLONE_IF_MISSING="yes"
DO_VERIFY="yes"
BUILD_DEB="no"
DEB_OUT_DIR="$HOME/procfd_deb_build"
DEB_MAINTAINER=""

# ----------------------------
# Helpers
# ----------------------------
log()  { echo "[INFO] $*"; }
warn() { echo "[WARN] $*"; }
die()  { echo "[ERROR] $*" >&2; exit 1; }

confirm_yes() {
  local prompt="$1"
  local confirmation
  read -r -p "$prompt (yes/NO): " confirmation
  [[ "$confirmation" == "yes" ]] || die "Operation cancelled by user."
}

run_or_die() {
  # Usage: run_or_die "description" command [args...]
  local desc="$1"; shift
  log "$desc"
  if ! "$@"; then
    die "Failed: $desc"
  fi
}

run_or_warn() {
  # Usage: run_or_warn "description" command [args...]
  local desc="$1"; shift
  log "$desc"
  if ! "$@"; then
    warn "Non-fatal check failed: $desc"
    return 0
  fi
}

ensure_in_home_tree() {
  # Require running from within HOME (~/ or subdir) to match the user requirement.
  if [[ "${PWD}" != "${HOME}"* ]]; then
    die "Please run this script from within your home directory tree (~/...). Current dir: ${PWD}"
  fi
}

ensure_not_root() {
  if [[ "$(id -u)" -eq 0 ]]; then
    die "Do not run this script as root. Run as a normal user with sudo privileges."
  fi
}

ensure_sudo_works() {
  command -v sudo >/dev/null 2>&1 || die "sudo is required but not found."
  run_or_die "Validating sudo access (you may be prompted for your password)" sudo -v
}

ensure_debian11_arm64() {
  local os_id os_version arch uname_m
  os_id="$(. /etc/os-release && echo "${ID:-}")"
  os_version="$(. /etc/os-release && echo "${VERSION_ID:-}")"
  arch="$(dpkg --print-architecture)"
  uname_m="$(uname -m)"

  [[ "$os_id" == "debian" ]] || die "Unsupported OS: expected Debian, got ID=$os_id"
  [[ "$os_version" == 11* ]] || die "Unsupported Debian version: expected 11.x (Bullseye), got VERSION_ID=$os_version"
  if [[ "$arch" != "arm64" && "$uname_m" != "aarch64" ]]; then
    die "Unsupported architecture: expected arm64/aarch64, got dpkg=$arch uname=$uname_m"
  fi
}

is_pkg_installed() {
  local pkg="$1"
  dpkg -s "$pkg" >/dev/null 2>&1
}

apt_install_if_missing() {
  local pkgs_to_install=()
  local pkg
  for pkg in "$@"; do
    if is_pkg_installed "$pkg"; then
      log "Package already installed: $pkg"
    else
      pkgs_to_install+=("$pkg")
    fi
  done

  if [[ "${#pkgs_to_install[@]}" -gt 0 ]]; then
    run_or_die "Updating apt package lists" sudo apt-get update
    run_or_die "Installing packages: ${pkgs_to_install[*]}" sudo apt-get install -y "${pkgs_to_install[@]}"
  fi
}

ensure_command() {
  # Usage: ensure_command <cmd> <apt-package>
  local cmd="$1"
  local pkg="$2"
  if ! command -v "$cmd" >/dev/null 2>&1; then
    apt_install_if_missing "$pkg"
  fi
  command -v "$cmd" >/dev/null 2>&1 || die "Tool '$cmd' still not available after installing '$pkg'."
}

TEMP_DIR=""
cleanup() {
  if [[ -n "${TEMP_DIR}" && -d "${TEMP_DIR}" ]]; then
    rm -rf "${TEMP_DIR}"
  fi
}
trap cleanup EXIT SIGINT SIGTERM

# ----------------------------
# Argument parsing
# ----------------------------
usage() {
  cat <<'EOF'
Usage: ./install-procfd-rk3588-debian11.sh [options]

Options:
  --src-dir PATH        Use an existing procfd source directory (default: ~/deshaw-procfd)
  --no-clone            Do not clone if source dir is missing (offline-friendly; will error instead)
  --skip-verify         Skip post-install verification commands
  --build-deb           Also build a simple .deb package under ~/procfd_deb_build/
  --deb-maintainer STR  Maintainer string for the .deb control file (e.g., "Name <email>")
  -h, --help            Show this help
EOF
}

while [[ "${#}" -gt 0 ]]; do
  case "$1" in
    --src-dir)
      shift
      [[ "${1:-}" ]] || die "--src-dir requires a path"
      SRC_DIR="$1"
      ;;
    --no-clone)
      DO_CLONE_IF_MISSING="no"
      ;;
    --skip-verify)
      DO_VERIFY="no"
      ;;
    --build-deb)
      BUILD_DEB="yes"
      ;;
    --deb-maintainer)
      shift
      [[ "${1:-}" ]] || die "--deb-maintainer requires a string"
      DEB_MAINTAINER="$1"
      ;;
    -h|--help)
      usage
      exit 0
      ;;
    *)
      die "Unknown option: $1"
      ;;
  esac
  shift
done

# ----------------------------
# Rust version helpers
# ----------------------------
parse_rustc_version() {
  # Outputs "MAJOR MINOR PATCH" or empty on failure
  if ! command -v rustc >/dev/null 2>&1; then
    return 1
  fi
  # rustc 1.75.0 (82e1608df 2023-12-21)
  local v
  v="$(rustc --version 2>/dev/null | awk '{print $2}')"
  [[ -n "$v" ]] || return 1
  local major minor patch
  major="$(echo "$v" | cut -d. -f1)"
  minor="$(echo "$v" | cut -d. -f2)"
  patch="$(echo "$v" | cut -d. -f3 | sed 's/[^0-9].*$//')"
  [[ -n "$major" && -n "$minor" && -n "$patch" ]] || return 1
  echo "$major $minor $patch"
}

rustc_is_new_enough_for_2021() {
  local major minor patch
  if ! read -r major minor patch < <(parse_rustc_version); then
    return 1
  fi
  if (( major > RUST_2021_MIN_MAJOR )); then
    return 0
  fi
  if (( major < RUST_2021_MIN_MAJOR )); then
    return 1
  fi
  # major == 1
  (( minor >= RUST_2021_MIN_MINOR ))
}

source_cargo_env_if_present() {
  if [[ -f "$HOME/.cargo/env" ]]; then
    # shellcheck disable=SC1090
    source "$HOME/.cargo/env"
  fi
  export PATH="$HOME/.cargo/bin:/usr/local/bin:$PATH"
}

# ----------------------------
# Steps
# ----------------------------
preflight_checks() {
  log "Starting procfd installation for Debian 11 (Bullseye) ARM64 (RK3588)"
  ensure_not_root
  ensure_in_home_tree
  ensure_debian11_arm64
  ensure_sudo_works

  apt_install_if_missing ca-certificates curl build-essential pkg-config

  log "Checking /proc mount options (hidepid may limit procfd visibility for non-root users)"
  if mount | grep -E ' on /proc ' >/dev/null 2>&1; then
    mount | grep -E ' on /proc '
    if mount | grep -E ' on /proc ' | grep -q 'hidepid='; then
      warn "Detected hidepid on /proc. Non-root procfd runs may not see other users' processes."
      warn "If you need system-wide visibility, run procfd with sudo."
    fi
  else
    warn "Could not find /proc mount line via mount(8). Continuing."
  fi
}

install_rust_toolchain_if_needed() {
  # We need a Rust compiler new enough for Rust 2021 (>= 1.56).
  # If rustc is missing or too old, install rustup stable.
  source_cargo_env_if_present

  if rustc_is_new_enough_for_2021; then
    log "Found rustc new enough for Rust 2021: $(rustc --version)"
    return 0
  fi

  if command -v rustc >/dev/null 2>&1; then
    warn "Found rustc, but it may be too old for Rust 2021: $(rustc --version)"
    warn "Will install/update Rust via rustup to ensure compatibility."
  else
    log "rustc not found; will install Rust via rustup."
  fi

  ensure_command curl curl

  TEMP_DIR="$(mktemp -d)"
  local rustup_init_bin="$TEMP_DIR/rustup-init"

  run_or_die "Downloading rustup-init (aarch64) to $rustup_init_bin" curl -Lfo "$rustup_init_bin" "$RUSTUP_INIT_URL_AARCH64"
  [[ -s "$rustup_init_bin" ]] || die "Downloaded rustup-init is empty or missing."
  chmod +x "$rustup_init_bin"

  confirm_yes "WARNING: This will execute the downloaded Rust installer '$rustup_init_bin' and install/update Rust in your home directory (~/.cargo, ~/.rustup). Continue?"

  run_or_die "Running rustup-init installer" "$rustup_init_bin" -y --no-modify-path

  source_cargo_env_if_present
  command -v rustup >/dev/null 2>&1 || die "rustup not found after installation."
  run_or_die "Installing stable Rust toolchain" rustup toolchain install stable
  run_or_die "Setting stable as default toolchain" rustup default stable

  run_or_die "Printing rustc version" rustc --version
  rustc_is_new_enough_for_2021 || die "rustc is still not new enough for Rust 2021 after rustup install."
}

ensure_source_tree() {
  if [[ -d "$SRC_DIR" && -f "$SRC_DIR/Cargo.toml" ]]; then
    log "Using existing source directory: $SRC_DIR"
    return 0
  fi

  if [[ "$DO_CLONE_IF_MISSING" != "yes" ]]; then
    die "Source directory not found at '$SRC_DIR' and cloning is disabled (--no-clone)."
  fi

  ensure_command git git

  log "Source directory not found. Cloning procfd from upstream into: $SRC_DIR"
  if [[ -e "$SRC_DIR" && ! -d "$SRC_DIR" ]]; then
    die "Path exists but is not a directory: $SRC_DIR"
  fi

  run_or_die "Cloning repository" git clone "$SRC_REPO_URL" "$SRC_DIR"
  [[ -f "$SRC_DIR/Cargo.toml" ]] || die "Clone completed but Cargo.toml not found in $SRC_DIR"
}

build_procfd() {
  log "Building procfd (release)"
  source_cargo_env_if_present
  command -v cargo >/dev/null 2>&1 || die "cargo not found; Rust toolchain not available."

  (cd "$SRC_DIR" && run_or_die "Running cargo build --release" cargo build --release)
  [[ -x "$SRC_DIR/target/release/procfd" ]] || die "Build did not produce expected binary at target/release/procfd"
}

install_procfd_binary() {
  local built_bin="$SRC_DIR/target/release/procfd"

  if [[ -e "$INSTALL_PATH" ]]; then
    confirm_yes "WARNING: '$INSTALL_PATH' already exists and will be overwritten. Continue?"
  fi

  run_or_die "Installing procfd to $INSTALL_PATH" sudo install -m 0755 "$built_bin" "$INSTALL_PATH"
  run_or_die "Verifying installed procfd is runnable" "$INSTALL_PATH" --help >/dev/null
}

verify_features() {
  if [[ "$DO_VERIFY" != "yes" ]]; then
    warn "Skipping verification as requested (--skip-verify)."
    return 0
  fi

  log "Running procfd verification commands (some are best-effort)..."

  # Must-pass basic execution
  run_or_die "Basic table output for current shell PID" bash -c "\"$INSTALL_PATH\" --pid \"$$\" >/dev/null"

  # Must-pass JSON output
  run_or_die "JSON output sanity check" bash -c "\"$INSTALL_PATH\" --pid \"$$\" --json | head -c 200 >/dev/null"

  # Must-pass pid-only mode
  run_or_die "PID-only mode sanity check" bash -c "\"$INSTALL_PATH\" --cmd '/bash/' --pid-only >/dev/null"

  # Must-pass process filters
  run_or_die "Filter by current user" bash -c "\"$INSTALL_PATH\" --user \"$(id -un)\" --pid-only >/dev/null"
  run_or_die "Filter by exact cmd (bash)" bash -c "\"$INSTALL_PATH\" --cmd bash --pid-only >/dev/null"

  # Must-pass FD type filters on this PID
  run_or_die "type=cwd" bash -c "\"$INSTALL_PATH\" --pid \"$$\" --type cwd >/dev/null"
  run_or_die "type=exe" bash -c "\"$INSTALL_PATH\" --pid \"$$\" --type exe >/dev/null"
  run_or_die "type=root" bash -c "\"$INSTALL_PATH\" --pid \"$$\" --type root >/dev/null"

  # Best-effort: mmaps (can be restricted by procfs permissions)
  run_or_warn "type=mmap (may be restricted; non-fatal)" bash -c "\"$INSTALL_PATH\" --pid \"$$\" --type mmap >/dev/null"

  # Best-effort: socket filters may return empty or be permission-sensitive depending on system state
  run_or_warn "socket-type tcp (no DNS) (non-fatal)" bash -c "\"$INSTALL_PATH\" --no-dns --socket-type tcp >/dev/null"
  run_or_warn "socket-domain unix (no DNS) (non-fatal)" bash -c "\"$INSTALL_PATH\" --no-dns --socket-domain unix >/dev/null"
  run_or_warn "port filter 22 (no DNS) (non-fatal)" bash -c "\"$INSTALL_PATH\" --no-dns --port 22 >/dev/null"

  # Best-effort: pipe endpoint display (timing-sensitive)
  log "Pipe endpoint display test (non-fatal; timing-sensitive)"
  (echo hi | cat >/dev/null) &
  sleep 0.1
  run_or_warn "pipe display via --cmd /cat/" bash -c "\"$INSTALL_PATH\" --cmd /cat/ >/dev/null"

  log "Verification complete."
}

build_deb_package() {
  if [[ "$BUILD_DEB" != "yes" ]]; then
    return 0
  fi

  ensure_command dpkg-deb dpkg

  local version=""
  if [[ -f "$SRC_DIR/Cargo.toml" ]]; then
    version="$(awk -F'"' '/^version = /{print $2; exit}' "$SRC_DIR/Cargo.toml" || true)"
  fi
  if [[ -z "$version" ]]; then
    version="1.3.0"
    warn "Could not parse version from Cargo.toml; defaulting to $version"
  fi

  if [[ -z "$DEB_MAINTAINER" ]]; then
    DEB_MAINTAINER="$(id -un) <$(id -un)@$(hostname)>"
  fi

  local pkg_root="$DEB_OUT_DIR/procfd_${version}_arm64"
  local deb_control_dir="$pkg_root/DEBIAN"
  local deb_bin_dir="$pkg_root/usr/local/bin"

  log "Building .deb package under: $DEB_OUT_DIR"
  mkdir -p "$deb_control_dir" "$deb_bin_dir"

  run_or_die "Copying procfd binary into package root" install -m 0755 "$SRC_DIR/target/release/procfd" "$deb_bin_dir/procfd"

  cat > "$deb_control_dir/control" <<EOF
Package: procfd
Version: $version
Section: utils
Priority: optional
Architecture: arm64
Maintainer: $DEB_MAINTAINER
Description: Fast lsof-like tool (procfd)
 procfd lists process file descriptors, sockets, pipes, and more.
EOF

  run_or_die "Building .deb via dpkg-deb" dpkg-deb --build "$pkg_root"
  log "Built package: ${pkg_root}.deb"
  log "To install it: sudo dpkg -i '${pkg_root}.deb'"
}

main() {
  preflight_checks
  install_rust_toolchain_if_needed
  ensure_source_tree
  build_procfd
  install_procfd_binary
  verify_features
  build_deb_package

  log "All done!"
  log "Try: $INSTALL_PATH --pid $$"
  log "If you need system-wide visibility, try: sudo $INSTALL_PATH --port 22"
}

main "$@"
