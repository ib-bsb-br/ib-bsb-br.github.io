#!/usr/bin/env pwsh
<#
.SYNOPSIS
Find the most relevant markdown post in a Jekyll _posts directory and copy its body to the clipboard.

.DESCRIPTION
This script accepts topic or keyword input, searches every markdown file under a Jekyll _posts
directory, scores each document using weighted lexical similarity across title, slug, tags,
filename, and body, selects the single best match, strips the Jekyll frontmatter, and copies
only the markdown body to the clipboard.

By default it downloads and caches:
  ib-bsb-br/ib-bsb-br.github.io
branch:
  main

It can also work against a local _posts directory via -LocalPostsPath.

This is heuristic lexical similarity, not embeddings-based semantic search.

.PARAMETER Query
One or more words or phrases used to search the _posts collection.

.PARAMETER Owner
GitHub owner. Default: ib-bsb-br

.PARAMETER Repo
GitHub repo. Default: ib-bsb-br.github.io

.PARAMETER Branch
GitHub branch. Default: main

.PARAMETER LocalPostsPath
Optional local path to the _posts directory. If supplied, download is skipped.

.PARAMETER CacheDir
Directory used to cache the downloaded repository snapshot.

.PARAMETER Refresh
Force a fresh download even if a cached copy exists.

.PARAMETER Top
How many top-ranked candidates to print. Default: 5

.PARAMETER PrintBody
Also print the chosen body to stdout.

.PARAMETER PassThru
Emit the selected result object to the pipeline.

.PARAMETER AsOSC52
If Set-Clipboard is available, prefer OSC52 clipboard transfer. Useful for some SSH workflows.

.EXAMPLE
pwsh -File ./fb.ps1 reassess

.EXAMPLE
pwsh -File ./fb.ps1 "treesheets hierarchy swap" -Verbose

.EXAMPLE
pwsh -File ./fb.ps1 "clipboard powershell linux" -LocalPostsPath ./_posts -PrintBody

.EXAMPLE
pwsh -File ./fb.ps1 "gtd mlo priority" -AsOSC52
#>

[CmdletBinding()]
param(
    [Parameter(Mandatory = $true, Position = 0, ValueFromRemainingArguments = $true)]
    [string[]]$Query,

    [string]$Owner = 'ib-bsb-br',
    [string]$Repo = 'ib-bsb-br.github.io',
    [string]$Branch = 'main',

    [string]$LocalPostsPath,

    [string]$CacheDir = $(
        if ($env:XDG_CACHE_HOME) {
            Join-Path $env:XDG_CACHE_HOME 'pwsh-post-search'
        }
        elseif ($HOME) {
            Join-Path $HOME '.cache/pwsh-post-search'
        }
        else {
            Join-Path ([System.IO.Path]::GetTempPath()) 'pwsh-post-search'
        }
    ),

    [int]$Top = 5,

    [switch]$Refresh,
    [switch]$PrintBody,
    [switch]$PassThru,
    [switch]$AsOSC52
)

Set-StrictMode -Version Latest
$ErrorActionPreference = 'Stop'

$script:StopWords = [System.Collections.Generic.HashSet[string]]::new([System.StringComparer]::OrdinalIgnoreCase)
@(
    'a','an','and','or','the','of','to','in','on','for','with','by','from','at',
    'is','are','be','as','this','that','these','those','it','its',
    'de','da','do','das','dos','e','ou','em','para','com','sem','por','um','uma'
) | ForEach-Object { $null = $script:StopWords.Add($_) }

function Get-NormalizedText {
    param([AllowNull()][string]$Text)

    if ([string]::IsNullOrWhiteSpace($Text)) {
        return ''
    }

    return ([regex]::Replace($Text.Trim(), '\s+', ' '))
}

function Get-NormalizedTokens {
    param([AllowNull()][string]$Text)

    if ([string]::IsNullOrWhiteSpace($Text)) {
        return @()
    }

    $matches = [regex]::Matches($Text.ToLowerInvariant(), '\p{L}[\p{L}\p{N}_-]*|\p{N}+')
    $tokens = New-Object System.Collections.Generic.List[string]

    foreach ($m in $matches) {
        $token = $m.Value
        if ($token.Length -lt 2) { continue }
        if ($script:StopWords.Contains($token)) { continue }
        $tokens.Add($token)
    }

    return @($tokens)
}

function New-TokenCountMap {
    param([AllowNull()][string]$Text)

    $map = @{}

    foreach ($token in (Get-NormalizedTokens -Text $Text)) {
        if ($map.ContainsKey($token)) {
            $map[$token]++
        }
        else {
            $map[$token] = 1
        }
    }

    return $map
}

function Get-MapCount {
    param(
        [hashtable]$Map,
        [string]$Key
    )

    if ($null -eq $Map) { return 0 }
    if ($Map.ContainsKey($Key)) { return [int]$Map[$Key] }
    return 0
}

function Split-JekyllDocument {
    param([Parameter(Mandatory = $true)][string]$RawText)

    $m = [regex]::Match($RawText, '(?s)\A---\r?\n(.*?)\r?\n---\r?\n?(.*)\z')

    if ($m.Success) {
        $frontMatter = $m.Groups[1].Value
        $body = $m.Groups[2].Value
    }
    else {
        $frontMatter = ''
        $body = $RawText
    }

    $body = $body -replace '\A(?:\r?\n)+', ''

    [pscustomobject]@{
        FrontMatter = $frontMatter
        Body        = $body
    }
}

function Parse-SimpleFrontMatter {
    param([AllowNull()][string]$FrontMatter)

    $meta = @{}

    if ([string]::IsNullOrWhiteSpace($FrontMatter)) {
        return $meta
    }

    $lines = $FrontMatter -split '\r?\n'
    $currentKey = $null

    foreach ($line in $lines) {
        if ($line -match '^\s*([A-Za-z0-9_-]+)\s*:\s*(.*?)\s*$') {
            $key = $Matches[1].ToLowerInvariant()
            $rawValue = $Matches[2].Trim()
            $currentKey = $key

            if (
                ($rawValue.StartsWith("'") -and $rawValue.EndsWith("'")) -or
                ($rawValue.StartsWith('"') -and $rawValue.EndsWith('"'))
            ) {
                $rawValue = $rawValue.Substring(1, $rawValue.Length - 2)
            }

            if ($rawValue -match '^\[(.*)\]$') {
                $inner = $Matches[1]
                $items = @()
                foreach ($piece in ($inner -split ',')) {
                    $v = $piece.Trim()
                    if (
                        ($v.StartsWith("'") -and $v.EndsWith("'")) -or
                        ($v.StartsWith('"') -and $v.EndsWith('"'))
                    ) {
                        $v = $v.Substring(1, $v.Length - 2)
                    }
                    if (-not [string]::IsNullOrWhiteSpace($v)) {
                        $items += $v
                    }
                }
                $meta[$key] = $items
            }
            elseif ([string]::IsNullOrWhiteSpace($rawValue)) {
                $meta[$key] = @()
            }
            else {
                $meta[$key] = $rawValue
            }

            continue
        }

        if ($null -ne $currentKey -and $line -match '^\s*-\s*(.+?)\s*$') {
            $item = $Matches[1].Trim()
            if (
                ($item.StartsWith("'") -and $item.EndsWith("'")) -or
                ($item.StartsWith('"') -and $item.EndsWith('"'))
            ) {
                $item = $item.Substring(1, $item.Length - 2)
            }

            if (-not ($meta[$currentKey] -is [System.Array])) {
                $meta[$currentKey] = @()
            }

            $meta[$currentKey] += $item
            continue
        }

        if ($line -notmatch '^\s') {
            $currentKey = $null
        }
    }

    return $meta
}

function Get-ClipboardBackend {
    $setClipboard = Get-Command Set-Clipboard -ErrorAction SilentlyContinue
    if ($setClipboard) {
        return 'Set-Clipboard'
    }

    foreach ($name in 'wl-copy', 'xclip', 'xsel') {
        if (Get-Command $name -ErrorAction SilentlyContinue) {
            return $name
        }
    }

    return $null
}

function Copy-TextToClipboard {
    param(
        [Parameter(Mandatory = $true)][string]$Text,
        [switch]$AsOSC52
    )

    $backend = Get-ClipboardBackend

    if (-not $backend) {
        throw "No clipboard backend found. Tried Set-Clipboard, wl-copy, xclip, and xsel."
    }

    switch ($backend) {
        'Set-Clipboard' {
            if ($AsOSC52) {
                Set-Clipboard -Value $Text -AsOSC52
                return 'Set-Clipboard -AsOSC52'
            }
            else {
                Set-Clipboard -Value $Text
                return 'Set-Clipboard'
            }
        }
        'wl-copy' {
            $Text | & (Get-Command wl-copy).Source
            return 'wl-copy'
        }
        'xclip' {
            $Text | & (Get-Command xclip).Source -selection clipboard
            return 'xclip'
        }
        'xsel' {
            $Text | & (Get-Command xsel).Source --clipboard --input
            return 'xsel'
        }
        default {
            throw "Unsupported clipboard backend: $backend"
        }
    }
}

function Get-RepoPostsDirectory {
    param(
        [string]$Owner,
        [string]$Repo,
        [string]$Branch,
        [string]$CacheDir,
        [switch]$Refresh,
        [string]$LocalPostsPath
    )

    if ($LocalPostsPath) {
        $resolved = (Resolve-Path -LiteralPath $LocalPostsPath).Path
        if (-not (Test-Path -LiteralPath $resolved -PathType Container)) {
            throw "LocalPostsPath is not a directory: $resolved"
        }
        Write-Verbose "Using local _posts directory: $resolved"
        return $resolved
    }

    $snapshotRoot = Join-Path $CacheDir "$Owner-$Repo-$Branch"
    $postsDir = Join-Path $snapshotRoot '_posts'

    if ($Refresh -and (Test-Path -LiteralPath $snapshotRoot)) {
        Write-Verbose "Removing cached snapshot: $snapshotRoot"
        Remove-Item -LiteralPath $snapshotRoot -Recurse -Force
    }

    if (Test-Path -LiteralPath $postsDir -PathType Container) {
        Write-Verbose "Using cached _posts directory: $postsDir"
        return $postsDir
    }

    New-Item -ItemType Directory -Path $CacheDir -Force | Out-Null
    New-Item -ItemType Directory -Path $snapshotRoot -Force | Out-Null

    $zipUrl = "https://codeload.github.com/$Owner/$Repo/zip/refs/heads/$Branch"
    $zipPath = Join-Path ([System.IO.Path]::GetTempPath()) "$Repo-$Branch-$([guid]::NewGuid().Guid).zip"
    $extractDir = Join-Path ([System.IO.Path]::GetTempPath()) "$Repo-$Branch-$([guid]::NewGuid().Guid)"

    try {
        Write-Verbose "Downloading repository archive from: $zipUrl"
        Invoke-WebRequest -Uri $zipUrl -OutFile $zipPath

        Write-Verbose "Expanding repository archive..."
        Expand-Archive -LiteralPath $zipPath -DestinationPath $extractDir -Force

        $root = Get-ChildItem -LiteralPath $extractDir -Directory | Select-Object -First 1
        if (-not $root) {
            throw "Expanded archive did not contain a repository root directory."
        }

        $sourcePostsDir = Join-Path $root.FullName '_posts'
        if (-not (Test-Path -LiteralPath $sourcePostsDir -PathType Container)) {
            throw "Expanded repository does not contain _posts at: $sourcePostsDir"
        }

        Write-Verbose "Copying _posts to cache..."
        Copy-Item -LiteralPath $sourcePostsDir -Destination $snapshotRoot -Recurse -Force

        if (-not (Test-Path -LiteralPath $postsDir -PathType Container)) {
            throw "Cached _posts directory was not created as expected: $postsDir"
        }

        return $postsDir
    }
    finally {
        if (Test-Path -LiteralPath $zipPath) {
            Remove-Item -LiteralPath $zipPath -Force -ErrorAction SilentlyContinue
        }
        if (Test-Path -LiteralPath $extractDir) {
            Remove-Item -LiteralPath $extractDir -Recurse -Force -ErrorAction SilentlyContinue
        }
    }
}

function Get-PostDocuments {
    param(
        [Parameter(Mandatory = $true)][string]$PostsDir,
        [Parameter(Mandatory = $true)][string]$Owner,
        [Parameter(Mandatory = $true)][string]$Repo,
        [Parameter(Mandatory = $true)][string]$Branch
    )

    $files = Get-ChildItem -LiteralPath $PostsDir -Recurse -File |
        Where-Object { $_.Extension -in '.md', '.markdown' } |
        Sort-Object -Property FullName

    $docs = @()

    $postsDirNormalized = $PostsDir.TrimEnd('/', '\')

    foreach ($file in $files) {
        try {
            Write-Verbose "Loading file: $($file.FullName)"

            $raw = Get-Content -LiteralPath $file.FullName -Raw -Encoding UTF8
            $parts = Split-JekyllDocument -RawText $raw
            $meta = Parse-SimpleFrontMatter -FrontMatter $parts.FrontMatter

            $title = ''
            if ($meta.ContainsKey('title') -and $null -ne $meta['title']) {
                if ($meta['title'] -is [System.Array]) {
                    $title = [string](($meta['title'] | ForEach-Object { [string]$_ }) -join ' ')
                }
                else {
                    $title = [string]$meta['title']
                }
            }

            $slug = ''
            if ($meta.ContainsKey('slug') -and $null -ne $meta['slug']) {
                if ($meta['slug'] -is [System.Array]) {
                    $slug = [string](($meta['slug'] | ForEach-Object { [string]$_ }) -join ' ')
                }
                else {
                    $slug = [string]$meta['slug']
                }
            }

            $tags = @()
            if ($meta.ContainsKey('tags') -and $null -ne $meta['tags']) {
                if ($meta['tags'] -is [System.Array]) {
                    $tags = @($meta['tags'] | ForEach-Object { [string]$_ })
                }
                else {
                    $tags = @([string]$meta['tags'])
                }
            }

            $relativeInsidePosts = $file.FullName.Substring($postsDirNormalized.Length).TrimStart('\', '/')
            $relativeInsidePosts = $relativeInsidePosts -replace '\\', '/'

            $repoPath = "_posts/$relativeInsidePosts"
            $blobUrl  = "https://github.com/$Owner/$Repo/blob/$Branch/$repoPath"
            $rawUrl   = "https://raw.githubusercontent.com/$Owner/$Repo/$Branch/$repoPath"

            $searchTitle = Get-NormalizedText -Text (($title, $slug, $file.BaseName) -join ' ')
            $searchTags  = Get-NormalizedText -Text ($tags -join ' ')
            $searchBody  = Get-NormalizedText -Text $parts.Body

            $titleCounts = New-TokenCountMap -Text $searchTitle
            $tagCounts   = New-TokenCountMap -Text $searchTags
            $bodyCounts  = New-TokenCountMap -Text $searchBody

            $tokenSet = @{}
            foreach ($k in $titleCounts.Keys) { $tokenSet[[string]$k] = $true }
            foreach ($k in $tagCounts.Keys)   { $tokenSet[[string]$k] = $true }
            foreach ($k in $bodyCounts.Keys)  { $tokenSet[[string]$k] = $true }

            $docs += [pscustomobject]@{
                Path        = $file.FullName
                FileName    = $file.Name
                RepoPath    = $repoPath
                BlobUrl     = $blobUrl
                RawUrl      = $rawUrl
                Title       = $title
                Slug        = $slug
                Tags        = $tags
                FrontMatter = $parts.FrontMatter
                Body        = $parts.Body
                SearchTitle = $searchTitle
                SearchTags  = $searchTags
                SearchBody  = $searchBody
                TitleCounts = $titleCounts
                TagCounts   = $tagCounts
                BodyCounts  = $bodyCounts
                TokenSet    = $tokenSet
            }
        }
        catch {
            Write-Warning "Skipping file due to parse/load failure: $($file.FullName)"
            Write-Warning $_.Exception.Message
        }
    }

    return @($docs)
}

function Measure-DocumentScore {
    param(
        [Parameter(Mandatory = $true)]$Document,
        [Parameter(Mandatory = $true)][string]$QueryText,
        [Parameter(Mandatory = $true)][string[]]$QueryTokens
    )

    $score = 0.0
    $normalizedQuery = (Get-NormalizedText -Text $QueryText).ToLowerInvariant()

    $titleText = $Document.SearchTitle.ToLowerInvariant()
    $tagsText  = $Document.SearchTags.ToLowerInvariant()
    $bodyText  = $Document.SearchBody.ToLowerInvariant()

    if ($normalizedQuery) {
        if ($titleText.Contains($normalizedQuery)) { $score += 70 }
        if ($tagsText.Contains($normalizedQuery))  { $score += 40 }
        if ($bodyText.Contains($normalizedQuery))  { $score += 15 }
    }

    $overlap = 0
    foreach ($token in $QueryTokens) {
        $titleHits = Get-MapCount -Map $Document.TitleCounts -Key $token
        $tagHits   = Get-MapCount -Map $Document.TagCounts   -Key $token
        $bodyHits  = Get-MapCount -Map $Document.BodyCounts  -Key $token

        if (($titleHits + $tagHits + $bodyHits) -gt 0) {
            $overlap++
        }

        $score += 12 * [Math]::Min($titleHits, 3)
        $score +=  8 * [Math]::Min($tagHits, 4)
        $score +=  1.5 * [Math]::Min($bodyHits, 25)
    }

    if ($QueryTokens.Count -gt 0) {
        $coverage = $overlap / $QueryTokens.Count
        $score += 12 * $coverage

        $union = $Document.TokenSet.Keys.Count + $QueryTokens.Count - $overlap
        if ($union -gt 0) {
            $score += 5 * ($overlap / $union)
        }
    }

    return [Math]::Round($score, 4)
}

function Get-RankedDocuments {
    param(
        [Parameter(Mandatory = $true)]$Documents,
        [Parameter(Mandatory = $true)][string]$QueryText,
        [Parameter(Mandatory = $true)][string[]]$QueryTokens
    )

    $ranked = foreach ($doc in $Documents) {
        $score = Measure-DocumentScore -Document $doc -QueryText $QueryText -QueryTokens $QueryTokens
        [pscustomobject]@{
            Score    = $score
            Document = $doc
        }
    }

    return @(
        $ranked | Sort-Object `
            @{ Expression = 'Score'; Descending = $true },
            @{ Expression = { $_.Document.RepoPath }; Descending = $false }
    )
}

$queryText = ($Query -join ' ').Trim()
if ([string]::IsNullOrWhiteSpace($queryText)) {
    throw "You must provide at least one topic or keyword."
}

$queryTokens = @(Get-NormalizedTokens -Text $queryText)
if ($queryTokens.Count -eq 0) {
    throw "After normalization, the query had no useful tokens left. Try more specific keywords."
}

Write-Verbose "Query text: $queryText"
Write-Verbose "Query tokens: $($queryTokens -join ', ')"

$postsDir = Get-RepoPostsDirectory `
    -Owner $Owner `
    -Repo $Repo `
    -Branch $Branch `
    -CacheDir $CacheDir `
    -Refresh:$Refresh `
    -LocalPostsPath $LocalPostsPath

$documents = @(Get-PostDocuments -PostsDir $postsDir -Owner $Owner -Repo $Repo -Branch $Branch)
if ($documents.Count -eq 0) {
    throw "No markdown files were found under: $postsDir"
}

Write-Verbose "Loaded $($documents.Count) markdown documents from _posts."

$ranked = @(Get-RankedDocuments -Documents $documents -QueryText $queryText -QueryTokens $queryTokens)
$best = $ranked | Select-Object -First 1

if (-not $best -or $best.Score -le 0) {
    throw "No meaningful match was found for query: $queryText"
}

$selected = $best.Document
$clipboardBackend = Copy-TextToClipboard -Text $selected.Body -AsOSC52:$AsOSC52

Write-Host ""
Write-Host "Chosen post:" -ForegroundColor Cyan
Write-Host "  Repo path : $($selected.RepoPath)"
Write-Host "  Title     : $($selected.Title)"
Write-Host "  Score     : $($best.Score)"
Write-Host "  Blob URL  : $($selected.BlobUrl)"
Write-Host "  Clipboard : $clipboardBackend"
Write-Host ""

$preview = $ranked | Select-Object -First ([Math]::Max(1, $Top))
Write-Host "Top candidates:" -ForegroundColor DarkCyan
foreach ($item in $preview) {
    $doc = $item.Document
    $displayTitle = if ([string]::IsNullOrWhiteSpace($doc.Title)) { '(no title)' } else { $doc.Title }
    Write-Host ("  {0,8}  {1}  [{2}]" -f $item.Score, $doc.RepoPath, $displayTitle)
}
Write-Host ""

if ($PrintBody) {
    Write-Output $selected.Body
}

if ($PassThru) {
    [pscustomobject]@{
        Query            = $queryText
        QueryTokens      = $queryTokens
        Score            = $best.Score
        RepoPath         = $selected.RepoPath
        Title            = $selected.Title
        Slug             = $selected.Slug
        Tags             = $selected.Tags
        BlobUrl          = $selected.BlobUrl
        RawUrl           = $selected.RawUrl
        ClipboardBackend = $clipboardBackend
        Body             = $selected.Body
    }
}
